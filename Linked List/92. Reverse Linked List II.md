# 92. Reverse Linked List II

Reverse Internal Linked List - 
[LeetCode 92](https://leetcode.com/problems/reverse-linked-list-ii/)


Reverse a linked list from position m to n. Do it in one-pass.

Note: 1 ≤ m ≤ n ≤ length of list.

Example:

Input: 1->2->3->4->5->NULL, m = 2, n = 4
Output: 1->4->3->2->5->NULL


```JavaScript
// iterative is better
var reverseBetween = function(curr, m, n) {
    if (m == n) return curr
    let count = 0
    let head = null
    let reversedEnd = null
    let foreword = null
    let prev = null
    while (curr) {
        if (count === 0) {
            head = curr
        }
        count += 1
        // check count
        if (count < m) {
            // step ahead
            prev = curr
            curr = curr.next
        }
        else if (count === m) {
            foreword = prev
            reversedEnd = curr
            prev = curr
            curr = curr.next
        }
        else if (count <= n) {
            if (count === m + 1) {
                prev.next = null        // manually delete the previous link
            }
            const next = curr.next
            curr.next = prev            // reverse link
            prev = curr
            curr = next
        }
        else break
    }
    var appendix = curr
    head = m === 1 ? prev : head
    reversedEnd.next = appendix
    if (m !== 1) {
        foreword.next = prev
    }
    return head
};
```
